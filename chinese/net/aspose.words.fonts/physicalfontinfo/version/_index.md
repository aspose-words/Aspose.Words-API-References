// 将 Aspose.Words 配置为从自定义文件夹获取字体，然后打印所有可用字体。

foreach (PhysicalFontInfo fontInfo in folderFontSource[0].GetAvailableFonts())

    Console.WriteLine("FullFontName  : {0}", fontInfo.FullFontName);
